CREATE TABLE `board` (
	`id` INT(10) NOT NULL AUTO_INCREMENT COMMENT '번호',
	`title` VARCHAR(100) NOT NULL COMMENT '제목' COLLATE 'utf8mb4_general_ci',
	`content` TEXT NOT NULL COMMENT '내용' COLLATE 'utf8mb4_general_ci',
	`read_cnt` INT(11) NOT NULL DEFAULT '0' COMMENT '조회수',
	`writer` VARCHAR(20) NOT NULL COMMENT '작성자' COLLATE 'utf8mb4_general_ci',
	`write_time` DATETIME NOT NULL COMMENT '작성일',
	`update_time` DATETIME NOT NULL COMMENT '수정일',
	PRIMARY KEY (`id`) USING BTREE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=23
;

CREATE TABLE `board_file` (
	`id` INT(10) NOT NULL AUTO_INCREMENT COMMENT '파일 번호',
	`board_no` INT(10) NOT NULL COMMENT '게시판 번호',
	`orgin_name` VARCHAR(50) NOT NULL COMMENT '원본 이름' COLLATE 'utf8mb4_general_ci',
	`random_name` VARCHAR(50) NOT NULL COMMENT '랜덤 이름' COLLATE 'utf8mb4_general_ci',
	`regdate` DATETIME NOT NULL COMMENT '등록 날짜',
	PRIMARY KEY (`id`) USING BTREE,
	INDEX `board_no` (`board_no`) USING BTREE,
	CONSTRAINT `FK_board_file_board` FOREIGN KEY (`board_no`) REFERENCES `test`.`board` (`id`) ON UPDATE NO ACTION ON DELETE CASCADE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;
